async function doAction(root, allDependencies, action = 'install') {
  typeof allDependencies === 'string'
    ? (allDependencies = [allDependencies])
    : null

    const command = 'npm'
    const args = [
      action,
      '--save',
      '--save-exact',
      '--loglevel',
      'error',
      ...allDependencies,
      '--cwd',
      root,
    ]

    child.on('close', resolve) // å®‰è£…æˆåŠŸåè§¦å‘resolve

    if (!fs.statSync(path).isDirectory()) {
