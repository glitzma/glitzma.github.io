# 柯理化

特点：

  • 参数复用： 如果相同的参数，在计算之后不需要再次重新传参计算
  • 提前返回： 多次调用，多次内部判断，可以将第一次返回的结果返回外部接收
  • 延迟执行： 避免重复去执行程序，等真正需要结果的时候再执行
  function add(){
      var _args = Array.prototype.slice.call(arguments)

      var _addr = function () {
        _args.push(...arguments)
        return _addr
      }

      // 当执行时隐式转换，并计算最终的值返回
      _addr.toString= function () {
        console.log(22)
        return _args.reduce(function (a,b) {
          return a + b;
        })
      }
      return _addr
    }

    console.log(add(1)(2)(3))

# bind 的实现


    Function.prototype.bind2 = function (context) {
      var self = this;
      var _args = [].slice.call(arguments, 1);

      return function () {
        var bindArgs = [].slice.call(arguments);

        self.apply(context, _args.concat(bindArgs));
      }
    }

    function bar(name, age) {
      console.log(name);
      console.log(age);
      console.log(this.value);
    }

    var foo = {
      value: 2
    }
    var bindFoo = bar.bind2(foo, 'mammmmmm');
    bindFoo('learn');
