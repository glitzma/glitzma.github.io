# 1. webpack编译生成js代码

// webpack编译生成活动js代码
// 组件预览
// 1、根据请求参数生成预览js代码；
// 2、然后通过webpack编译；
// 3、将生成的代码上传后端/acplat/config/bosupload接口,生成cdn地址

var child_exec = require('child_process').exec;
var fs = require('fs');
var path = require('path');
var ejs = require('ejs');
var _ = require('underscore');
var webpack = require('webpack');
var MemoryFS = require('memory-fs');
var webpackConfig = require('./webpack.config.js');

// 编译固定活动入口js文件和vue文件
const indexJsTpl = fs.readFileSync(__dirname + '/index.js.tpl', 'utf8');
const indexVueTpl = fs.readFileSync(__dirname + '/index.vue.tpl', 'utf8');
const indexJsCompile = ejs.compile(indexJsTpl);
const indexVueCompile = ejs.compile(indexVueTpl);

let mfs = new MemoryFS();

function generatePreCompileCode(req) {
    // 创建该活动的入口文件
    let destDirName = `web_${Date.now()}`;
    let dirname = `${__dirname}/${destDirName}`;
    let filename = `${dirname}/index.js`;
    fs.mkdirSync(dirname);
    fs.mkdirSync(`${dirname}/build`);

    let content =  indexJsCompile({activityConfig: req.body.activityConfig});

    fs.writeFileSync(filename, content);

    let entryFileName = `${dirname}/index.vue`;
    if (typeof req.body.compList  === 'string') {
        req.body.compList = [req.body.compList];
    }
    let compNum = req.body.compList.length || 0;
    let compStr = '';
    for (let i = 0; i < compNum; i++) {
        let comp = req.body.compList[i];
        compStr += `'${comp}': require('${comp}/index.vue'),`;
    }

    let vueContent = indexVueCompile({
        activityComponents: req.body.activityComponents,
        componentData: req.body.componentData,
        childCompList: compStr
    });

    fs.writeFileSync(entryFileName, vueContent);
    // 编译vue代码
    return webpackCompile(req, dirname, destDirName);
}

function webpackCompile(req, codeDir, destDirName) {
    // webpack在编译时会修改配置对象，所以为了多次使用需要保存原始配置
    let webpackNewConfig = _.clone(webpackConfig);
    // 修改代码编译目录
    webpackNewConfig.output.path = `${codeDir}/build`;
    webpackNewConfig.context = codeDir;

    let compiler = webpack(webpackNewConfig);

    compiler.outputFileSystem = mfs;

    let promise = new Promise((resolve, reject) => {

        compiler.run(function(err, stats) {
            if (err) {
                console.error('编译vue出错：', err);
                reject(err);
                return;
            }

            const info = stats.toJson();

            if (stats.hasErrors()) {
                console.error(info.errors);
                reject(info.errors);
                return;
            }

            let result = {};
            // 读取文件内容
            result.jsCode = mfs.readFileSync(`${codeDir}/build/bundle.js`, 'utf8');
            result.cssCode = mfs.readFileSync(`${codeDir}/build/bundle.css`, 'utf8');

            // 临时放到本地
            // fs.writeFileSync(`${codeDir}/build/bundle.js`, result.jsCode);
            // fs.writeFileSync(`${codeDir}/build/bundle.css`, result.cssCode);


            // 返回编译后的生成结果
            resolve(result);
            // 删除代码目录
            console.info('删除代码编译目录：', destDirName);
            // child_exec(`rm -rf ${destDirName}`, {
            //     cwd: path.resolve(__dirname)
            // }, error => {
            //     if (error) {
            //         console.error(`删除生成代码目录${codeDir}出错`);
            //     }
            // });
        });
    });
    return promise;
}

module.exports = {
    generatePreCompileCode: generatePreCompileCode
};

# 关于Multer

Multer 是一个 node.js 中间件，用于处理 multipart/form-data 类型的表单数据，它主要用于上传文件。它是写在 busboy 之上非常高效。
