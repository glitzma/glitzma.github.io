数组 的map filter find

let arr = [1,5,37,9]

arr.filter //查找某一项 返回新数组

arr.map //替换项  用map做映射

arr.find(function(item,index){        //找到具体的某一项
    return item.toSTring().indexOf(5) > -1
})

some 找true 找到true 后停止，返回true 找不到，返回false

arr.some(function(item,index){
    return item.toSTring().indexOf(5) > -1;
})

every 找false 找到false后停止 返回false

reduce 4个参数 ，返回的是叠加后的结果。原数组不变，回调函数返回结果

[1,2,3,4,5].reduce(function(prev,next,index,item){
console.log(arguments);
//[1, 2, 1, Array(5), callee: ƒ, Symbol(Symbol.iterator): ƒ]

// prev 代表数组的第一项，next代表数组的第二项
//第二次prev是undefined ,next是数组的第三项 （因为prev没有返回值）
})

//可以用来 return prev+next;

[{price:10,count:1},{price:15,count:10},{price:30,count:5}].reduce(function(prev,next,index,item){
    return prev + next.price * next.count;
},0);
//后面加0 可以设置初始值

//可以进行数组的拼接

join

[1,2,[4,5],7].join(',');
// "1,2,4,5,7"
